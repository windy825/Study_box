#### 뉴스

```python
alpha = 'abcdefghijklmnopqrstuvwxyz'


def solution(str1, str2):
    # 소문자 변환
    str1 = str1.lower()
    str2 = str2.lower()
    
    # 딕셔너리로 개수만 비교
    first = {}
    second = {}
    
    # 2개 단위로 쪼개서 딕셔너리에 개수 체크
    for i in range(len(str1)-1):
        if str1[i:i+2].isalpha():
            first.setdefault(str1[i:i+2], 0)
            first[str1[i:i+2]] += 1
    for j in range(len(str2)-1):
        if str2[j:j+2].isalpha():
            second.setdefault(str2[j:j+2], 0)
            second[str2[j:j+2]] += 1
            
    # 각 집합의 크기
    A = sum(first[key] for key in first)
    B = sum(second[key] for key in second)
    
    # B가 0이 아니기만 하면 됨
    if B:
        # 각 원소의 개수 중 합이 제일 작은 녀석으로(교집합)
        C = sum(min(second.get(key, 0), first[key]) for key in first)
        answer = int(C / (A+B-C) * 65536)	# 합집합 : 각 집합의 합 - 교집합
    
    # B가 0인 경우(A, B 둘 다 0)
    else:
        answer = 65536
    return answer
```

#### 여행경로

```python
total = []		# 모든 경우의 수
visit = []		# 방문 체크
L = 0			# 티켓 개수
T = []			# tickets 리스트

def dfs(arr):
    if len(arr) > L:	# 완성된 arr는 L + 1 만큼의 길이
        total.append(arr)
        return
    else:
        for j in range(L):
            if not visit[j] and arr[-1] == T[j][0]:		# 방문 안 하고 출발가능한 경우
                visit[j] = 1
                dfs(arr + [T[j][1]])
                visit[j] = 0

def solution(tickets):
    global visit, L, T
    T = tickets
    L = len(tickets)
    visit = [0] * L
    for i in range(L):
        if T[i][0] == 'ICN':
            visit[i] = 1
            dfs(T[i])
            visit[i] = 0
    answer = sorted(total)[0]
    return answer
```
