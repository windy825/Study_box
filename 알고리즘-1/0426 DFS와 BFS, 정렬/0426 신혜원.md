#### 타겟 넘버

```python
answer = 0

def solution(numbers, target):
    global N
    N = len(numbers)
    
    # dfs 2가지 경우(+, -)
    dfs(0, numbers[0], numbers, target)
    dfs(0, -numbers[0], numbers, target)
    return answer

def dfs(k, summ, numbers, target): #단계, 합, lst, target
    global answer
    if k == N-1:
        # summ과 target이 같을 때마다 정답을 1개씩 늘려줌
        if summ == target:
            answer += 1
        return
    else:
        # 2가지 경우(+, -)
        dfs(k+1, summ+numbers[k+1], numbers, target)
        dfs(k+1, summ-numbers[k+1], numbers, target)
```

#### H-index

```python
def solution(citations):
    N = len(citations)
    
    citations.sort()
    h_index = 0
    # 정렬 후 역순으로 순회
    for i in range(N - 1, -1, -1):
        # h_index 값을 올려가며, citations 값이 h_index보다 같거나 작을 때 break
        if citations[i] <= h_index:
            answer = h_index
            break
            h_index += 1
            # [1, 1, 1] 같은 경우는 h_indedx가 N값과 같음
            else:
                answer = N

    return answer
```
