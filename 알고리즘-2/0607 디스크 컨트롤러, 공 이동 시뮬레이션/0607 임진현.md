#### 디스크 컨트롤러

```python
import heapq

def solution(jobs):
    cnt, start, now = 0, -1, 0
    answer = 0
    min_heap = []

    while cnt != len(jobs):
        for i in jobs:
            if start < i[0] <= now:
                heapq.heappush(min_heap, [i[1], i[0]])
        
        if min_heap:
            span, starttime = heapq.heappop(min_heap)
            start = now
            now += span
            answer += now - starttime
            cnt += 1
        
        else:
            now += 1 

   
    return int(answer / len(jobs))
```
