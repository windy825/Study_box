#### 삼각 달팽이

```python
# 방향 키
di, dj = [1, 0, -1],[0, 1, -1] 

def solution(n):
    answer = []
    arr = [[0]*n for _ in range(n)]

    # 초기값 설정
    ci, cj, num, k = 0,0,0,0
    arr[ci][cj] = 1

    # 종료 조건 값 구하기
    for i in range(1, n+1):
        num += i

    while True:
        # 종료 조건
        if arr[ci][cj] == num:
            break

        # 다음 칸으로 이동
        ni,nj = ci+di[k], cj+dj[k]
        if 0<=ni<n and 0<=nj<n and arr[ni][nj]==0:
            arr[ni][nj] = arr[ci][cj]+1
            ci,cj = ni,nj
        
        # 배열 범위 초과 시 방향 키 설정
        else:
            k = (k+1)%3
	# answer에 담기
    for i in range(n):        
        for j in range(n):
            if arr[i][j]:
                answer.append(arr[i][j])
    return answer
```
